
 select * from (
 SELECT t3.areaNo,
       cityNo ,
       staffId,
       staffName,
       attrValue,
       zwType,
       responsibleArea,
       toSysDate,
       ydTotal,
       zdkdTotal,
       qwWiFiServerTotal,
       qwWiFiTerminalTotal,
       tygqTotal,
       tykjTotal,
       znjjTotal,
       znjjFTTRTotal,
       fttrYfTotal,
       fttrbTotal,
       sqznzwTotal,
       rhzk151Total,
       tyzpTotal,
       ydTotal + zdkdTotal + qwWiFiServerTotal + qwWiFiTerminalTotal
         + tygqTotal + tykjTotal + znjjTotal + znjjFTTRTotal
         + fttrYfTotal + sqznzwTotal + fttrbTotal + rhzk151Total + tyzpTotal AS qywTotal,
       ROUND(
         (zdkdTotal * 0.5)
         + (qwWiFiTerminalTotal * 0.25)
         + znjjFttrTotal
         + (sqznzwTotal * 0.25 * 1.5)
         + (fttrbTotal * 1.5), 1) AS zslTotal,
       ROUND(
         ((qwWiFiTerminalTotal * 0.25)
          + znjjFttrTotal
          + (sqznzwTotal * 0.25 * 1.5)
          + (fttrbTotal * 1.5)) * 60, 2) AS jzPoint,
       COALESCE(t2.yyPoint, 0)  AS yyPoint,
       COALESCE(t2.fzPoint, 0)  AS fzPoint,
       COALESCE(t2.sumPoint, 0) AS sumPoint,
       ROUND(
         COALESCE(t2.fzPoint, 0)
         + (((qwWiFiTerminalTotal * 0.25)
            + znjjFttrTotal
            + (sqznzwTotal * 0.25 * 1.5)
            + (fttrbTotal * 1.5)) * 60), 2) AS pjPoint
FROM
  (SELECT areaNo,
          cityNo,
          staff_num   AS staffId,
          staff_name  AS staffName,
          zwType,
          responsibleArea,
          toSysDate,
          CASE
              WHEN attr_value = '10'  THEN '一线装维'
              WHEN attr_value = '20'  THEN '装维班长'
              WHEN attr_value = '50'  THEN '装维经营承包'
              WHEN attr_value = '30'  THEN '客调'
              WHEN attr_value = '40'  THEN '代理商装维'
              WHEN attr_value = '60'  THEN '客户工程师'
              WHEN attr_value = '70'  THEN '客户经理'
              WHEN attr_value = '80'  THEN '其他人员'
              WHEN attr_value = '200' THEN '装维门店'
              WHEN attr_value = '72'  THEN '智家工程师-家庭DICT'
              WHEN attr_value = '73'  THEN '智家工程师-校园'
              ELSE ''
          END AS attrValue,
          SUM(COALESCE(yd, 0))         AS ydTotal,
          SUM(COALESCE(kd, 0))         AS zdkdTotal,
          SUM(COALESCE(wifi_fw, 0))    AS qwWiFiServerTotal,
          SUM(COALESCE(wifi_zd, 0))    AS qwWiFiTerminalTotal,
          SUM(COALESCE(itv, 0))        AS tygqTotal,
          SUM(COALESCE(tysl, 0))       AS tykjTotal,
          SUM(COALESCE(qwzn, 0))       AS znjjTotal,
          SUM(COALESCE(qwzn_fttr, 0))  AS znjjFTTRTotal,
          SUM(COALESCE(fttr_yf, 0))    AS fttrYfTotal,
          SUM(COALESCE(fttrb, 0))      AS fttrbTotal,
          SUM(COALESCE(sqznzw, 0))     AS sqznzwTotal,
          SUM(COALESCE(rhzk151, 0))    AS rhzk151Total,
          SUM(COALESCE(tyzp, 0))       AS tyzpTotal
   FROM
     (SELECT kd,
             yd,
             itv,
             wifi_fw,
             wifi_zd,
             qwzn_fttr,
             qwzn,
             tysl,
             fttr_yf,
             sqznzw,
             fttrb,
             rhzk151,
             tyzp,
             b.staff_num,
             b.attr_value,
             b.staff_name,
             b.area_no  AS areaNo,
             b.city_no  AS cityNo,
             b.zw_type  AS zwType,
             b.responsible_area AS responsibleArea,
             b.to_sys_date      AS toSysDate
      FROM
        (SELECT staff_id,
                month_id,
                attr_value_sx AS attr_value,
                staff_num,
                staff_name,
                area_no,
                city_no,
                zw_type,
                responsible_area,
                to_sys_date
         FROM iceberg.dwd.dwd_d_mss_hr_dim_staff_zw10_zw zw
         LEFT JOIN iceberg.dwd.dwd_d_mss_hr_emp_employee_ext_month emp
                ON CAST(zw.emp_id AS bigint) = CAST(emp.emp_id AS bigint)
               AND emp.month = CAST(zw.month_id AS varchar)
               AND CAST(emp.ext1 AS varchar) = '1'
               AND CAST(emp.is_confirm AS varchar) = '1'
               and  emp.p_acct_day   = '20260706'
               and zw.p_acct_day   = '20260706'
         WHERE month_id = '202607' 
     
           AND zw.is_off = '1') b
      LEFT JOIN iceberg.dwd.dwd_d_mss_hr_dwv_d_smr_user_list_result_merge a
             ON a.dvlp_staff_id = b.staff_id
            AND CAST(a.new_date AS date) >= DATE '2026-07-01'
            AND CAST(a.new_date AS date) <= DATE '2026-07-06'
            and a.p_acct_day = '20260706'
      WHERE b.month_id = '202607') t1
   GROUP BY areaNo,
            cityNo,
            staff_num,
            staff_name,
            attr_value,
            zwType,
            responsibleArea,
            toSysDate) t3
LEFT JOIN
  (SELECT staff_num,
          COALESCE(SUM(CAST(fz_point AS double)), 0) AS fzPoint,
          COALESCE(SUM(CAST(yy_point AS double)), 0) AS yyPoint,
          SUM(COALESCE(CAST(fz_point AS double), 0)
            + COALESCE(CAST(yy_point AS double), 0)) AS sumPoint
   FROM iceberg.dwd.dwd_d_mss_hr_dwv_d_hrt_zw_point_zw_merge
   WHERE attr_value IS NOT NULL
     AND acct_day >= '20260701'
     AND acct_day <= '20260706'
     and p_acct_day = '20260706'
   GROUP BY staff_num) t2
       ON t3.staffId = t2.staff_num
 
 )
 where staffName = '刘志昊'
